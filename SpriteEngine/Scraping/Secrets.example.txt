// Template for SpriteEngine/Scraping/Secrets.swift
//
// Setup:
//   1. Register a free user account at https://www.screenscraper.fr
//   2. Post in the dev forum (forumsujets.php?frub=12) requesting `devid`/`devpassword`
//   3. Copy this file to `Secrets.swift` next to it (the real file is gitignored)
//   4. Fill in the values below with what was issued to you

import Foundation

enum ScreenScraperSecrets {
    static let devID:          String = ""
    static let devPassword:    String = ""
    static let debugPassword:  String = ""
    static let softname:       String = "Sprite Engine"

    static var isConfigured: Bool {
        !devID.isEmpty && !devPassword.isEmpty
    }
}
